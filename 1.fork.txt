#include <stdio.h>
#include <unistd.h>
#include <sys/types.h>
#include <sys/wait.h>

int main() {
    pid_t pid;
    int status;

    printf("Enter a command to execute: ");
    char command[100];
    scanf("%s", command);

    pid = fork();
    if (pid == -1) {
        perror("fork error");
        return 1;
    } else if (pid == 0) {
        // child process
        execvp(command, NULL);
        perror("exec error");
        return 1;
    } else {
        // parent process
        wait(&status);
        printf("Child process exited with status %d.\n", status);
        return 0;
    }
}
