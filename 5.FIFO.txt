#include <stdio.h>

int main() {
    int page_frames, page_count, page_faults = 0;
    printf("Enter the number of page frames: ");
    scanf("%d", &page_frames);
    printf("Enter the number of pages: ");
    scanf("%d", &page_count);
    int pages[page_count];
    printf("Enter the page reference string: ");
    for (int i = 0; i < page_count; i++) {
        scanf("%d", &pages[i]);
    }
    printf("\n");

    int frames[page_frames];
    for (int i = 0; i < page_frames; i++) {
        frames[i] = -1; // set all frames to -1 (empty)
    }

    int frame_index = 0; // index to keep track of which frame to replace
    for (int i = 0; i < page_count; i++) {
        int page = pages[i];
        int found = 0;
        for (int j = 0; j < page_frames; j++) {
            if (frames[j] == page) { // page already present in frame
                found = 1;
                break;
            }
        }

        if (!found) { // page not present in frame, replace the oldest page in frame
            frames[frame_index] = page;
            frame_index = (frame_index + 1) % page_frames;
            page_faults++;
        }

        printf("Page reference: %d, Page frames: ", page);
        for (int j = 0; j < page_frames; j++) {
            printf("%d ", frames[j]);
        }
        printf("\n");
    }

    printf("\nTotal page faults: %d\n", page_faults);
    return 0;
}
